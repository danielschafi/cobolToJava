# Java
Install java runtime and compiler


cd ~
wget https://download.oracle.com/java/21/archive/jdk-21.0.6_linux-x64_bin.tar.gz
tar -xzf jdk-21.0.6_linux-x64_bin.tar.gz
mv jdk-21.0.6 ~/jdk-21

echo 'export JAVA_HOME=$HOME/jdk-21' >> ~/.bashrc
echo 'export PATH=$JAVA_HOME/bin:$PATH' >> ~/.bashrc
source ~/.bashrc

java -version
javac -version



# Antlr
Install COBOL85 grammar files etc. since we cant use apt-get ti usbtak us

1. Download ANTLR jar
wget https://www.antlr.org/download/antlr-4.13.1-complete.jar -O antlr-4.13.1-complete.jar

2. Set up an alias , replace ~/cobolToJava/ with wherever you saved the JAR
alias antlr4='java -jar ~/cobolToJava/antlr-4.13.1-complete.jar'
alias grun='java -jar ~/cobolToJava/antlr-4.13.1-complete.jar org.antlr.v4.gui.TestRig'

Download the COBOL85 grammar files
wget https://raw.githubusercontent.com/antlr/grammars-v4/master/cobol85/Cobol85.g4
wget https://raw.githubusercontent.com/antlr/grammars-v4/master/cobol85/Cobol85Preprocessor.g4



3. Run ANTLR to generate the parser
java -jar antlr-4.13.1-complete.jar -Dlanguage=Python3 Cobol85.g4 Cobol85Preprocessor.g4

3. Install the ANTLR Python runtime
pip install antlr4-python3-runtime
